  ```JavaScript
  /**
   * 导出
   */
  async onDerive() {
    wxapp.showLoading('文件生成中...')
    const [openId, token] = wxapp.getStorageSyncByArray(['openId', 'accessToken'])
    const header = {
      Authorization: token ? `Bearer ${token}` : undefined,
      openId,
    }
    wx.request({
      url: _CONFIG.urlPrefix + api.couponApi.recordsDownload,
      method: 'POST',
      header: {
        ...header,
        'Content-Type': 'application/json;charset=UTF-8',
      },
      responseType: 'arraybuffer',
      data: {
        max: 10,
        pageNumber: this.data.pageNumber,
        ...this.data.paramObj,
      },
      success(res: anyKey) {
        if (res && res.data) {
          const filePath = `${wx.env.USER_DATA_PATH}/write-off-record.xls`
          wx.getFileSystemManager().writeFile({
            filePath,
            data: res.data,
            encoding: 'binary', // 二进制
            success() {
              wx.openDocument({
                filePath,
                showMenu: true,
                fail(_err) {
                  console.error('文件保存失败，请重试')
                  wxapp.hideLoading()
                },
              })
            },
            fail(_err) {
              console.error('文件保存失败，请重试')
              wxapp.hideLoading()
            },
          })
        }
      },
      fail(_err) {
        wx.showToast({
          title: '文件导出失败，请重试',
          icon: 'none',
        })
        wxapp.hideLoading()
      },
    })
  }
```
